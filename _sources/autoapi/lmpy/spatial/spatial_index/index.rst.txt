      :param att_dict: A dictionary of attributes to store in the lookup table.

      :param x: The x coordinate to search for.
      :type x: numeric
      :param y: The y coordinate to search for.
      :type y: numeric

      :returns: A dictionary of index hits for the search.
      :rtype: dict
